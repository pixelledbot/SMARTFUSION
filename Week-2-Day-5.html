    Exploratory Data Analysis (EDA) is the process of analyzing datasets to summarize their main characteristics, often using visual methods. EDA is crucial for understanding the structure of your data before modeling.

    • Identify missing values, outliers, and duplicates

    • Understand distributions and relationships using plots

  • Use Wokwi to simulate ESP32 + Sensor
  • Send data to ThingSpeak using HTTP
  • Visualize data using embedded charts
  • ESP32 Code (Wokwi)

    This code reads a simulated sensor value and sends it to ThingSpeak:

    
#include <WiFi.h>
#include <HTTPClient.h>

const char* ssid = "YOUR_WIFI";
const char* password = "YOUR_PASS";
const char* server = "http://api.thingspeak.com/update?api_key=YOUR_API_KEY";

void setup() {
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    String url = String(server) + "&field1=" + String(random(25, 40));
    http.begin(url);
    int httpCode = http.GET();
    http.end();
  }
  delay(15000);
}
